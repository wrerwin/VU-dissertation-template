\def\@makechapterhead#1{%
	{\parindent \z@ \centering
	\normalfont
	\ifnum \c@secnumdepth >\m@ne
	\if@mainmatter
	\@chapapp\space \thechapter
	\par\nobreak
	\vskip 20\p@
	\fi
	\fi
	\interlinepenalty\@M
	#1\par\nobreak
	\vskip 40\p@
	}}
	
\def\@schapter#1{\if@twocolumn
	\@topnewpage[\@makeschapterhead{#1}]%
	\else
	\@makeschapterhead{#1}%
	\@afterheading
	\fi
	}
	
\def\@makeschapterhead#1{
	{\parindent \z@ \centering
	\normalfont
	\interlinepenalty\@M
	#1\par\nobreak
	\vskip 10\p@
	}}
